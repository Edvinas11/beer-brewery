<beer> ::= <name> <type> <alcohol-content> <ingredients>

<name> ::= "Pale Ale" | "Guinness"

<type> ::= "Lager" | "Ale" | "Stout"

<alcohol-content> ::= <percentage>

<percentage> ::= <number> "%"

<number> ::= "0" | "1" | "2" | ... | "9"

<ingredients> ::= <ingredient> | <ingredient> <ingredients>

<ingredient> ::= "Malt" | "Hops" | "Yeast" | "Water"

<recipe> ::= <type> <ingredients> <process>

<process> ::= "Mash" <time> "Boil" <time> "Ferment" <time> "Condition" <time>

<time> ::= "30 minutes" | "1 hour" | "2 weeks" | "1 month"

<brew-command> ::= "brewBeer" <recipe>

<add-ingredient-command> ::= "addIngredient" <ingredient> <recipe>

<ferment-command> ::= "ferment" <beer> <time-period>

<condition-command> ::= "condition" <beer> <time-period>

<time-period> ::= "2 weeks" | "1 month" | "6 months"
